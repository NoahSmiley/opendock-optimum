6) AUTHN/AUTHZ DETAILS
--------------------------------------------------------------------
Password rules:
- min 8 chars, at least 1 letter and 1 number. Use zod refinement.
- Store bcrypt hash with cost 12–13. Never store plaintext.

Session cookie:
- Name: "od.sid"
- httpOnly, Secure (on HTTPS), SameSite=Lax
- Expires: 30 days; rotate on each login; revoke on logout
- Value: random 32‑byte token (base64url). Store only its SHA‑256 hash in DB.
- CSRF: issue "od.csrf" non‑httpOnly cookie with a random token. Require header "x-csrf-token" to match cookie value for all mutating requests. Exempt GET and SSE.

RBAC v1:
- Roles: owner>admin>member>viewer. Scope project‑wide for now.
- Middleware withRole('admin' | 'owner') for sensitive routes.

Boards & Tickets
- GET    /projects/:id/board               → 200 { board, columns, tickets }
- POST   /projects/:id/board/columns       → 201 Column
- PATCH  /boards/:boardId/columns/:colId   → 200 Column
- POST   /boards/:boardId/tickets          → 201 Ticket
- PATCH  /tickets/:id                      → 200 Ticket
- POST   /tickets/:id/comments             → 201 Comment
- POST   /boards/:boardId/reorder          → 204 (columns & tickets reorders)

Builds & Deployments
- POST   /projects/:id/builds              → 201 Build (status=queued)
- GET    /projects/:id/builds              → 200 Build[]
- GET    /builds/:id                       → 200 Build
- POST   /builds/:id/retry                 → 201 Build
- POST   /builds/:id/deploy                → 201 Deployment (queued)
- GET    /deploys/:id                      → 200 Deployment

Streaming (SSE)
- GET /sse/board/:boardId                  → event: "board:update" JSON payloads
- GET /sse/build/:buildId/logs             → event: "log:append" { lines: string[] }
- Always set headers: Content-Type: text/event-stream, Cache-Control: no-cache

Errors
- Error JSON shape: { error: { code: string; message: string; details?: any } }
- Use proper status codes; never leak stack traces in production.

--------------------------------------------------------------------

   - take queued builds → running → emit "build:started"

// apps/backend/src/auth/passwords.ts
import bcrypt from 'bcryptjs';
export async function hashPassword(pw: string) { return bcrypt.hash(pw, 12); }
export async function verifyPassword(pw: string, hash: string) { return bcrypt.compare(pw, hash); }

// apps/backend/src/auth/sessions.ts
import { randomBytes, createHash } from 'crypto';
import { prisma } from '../dal/sql/client';
const COOKIE = 'od.sid';
export function newToken() { return randomBytes(32).toString('base64url'); }
export function tokenHash(token: string) { return createHash('sha256').update(token).digest('base64url'); }
export async function createSession(userId: string, ua?: string, ip?: string) {
  const token = newToken();
  const hash = tokenHash(token);
  const expiresAt = new Date(Date.now() + 1000*60*60*24*30);
  await prisma.session.create({ data: { userId, tokenHash: hash, userAgent: ua, ip, expiresAt } });
  return { token, expiresAt };
}

// apps/backend/src/auth/middleware.ts (sketch)
import type { Request, Response, NextFunction } from 'express';
import { prisma } from '../dal/sql/client';
import { tokenHash } from './sessions';
export async function attachUser(req: Request, _res: Response, next: NextFunction) {
  const token = req.cookies?.['od.sid']; if (!token) return next();
  const sess = await prisma.session.findUnique({ where: { tokenHash: tokenHash(token) }, include: { user: true } });
  if (sess && sess.expiresAt > new Date()) (req as any).user = sess.user;
  return next();
}
export function authRequired(req: Request, res: Response, next: NextFunction) {
  if (!(req as any).user) return res.status(401).json({ error: { code: 'UNAUTHENTICATED', message: 'Login required.' }});
  next();
}

// apps/backend/src/routes/auth.routes.ts (sketch)
router.post('/auth/register', zodBody(RegisterSchema), async (req, res) => {
  const { email, password, displayName } = req.body;
  const existing = await prisma.user.findUnique({ where: { email } });
  if (existing) return res.status(409).json({ error: { code: 'EMAIL_TAKEN', message: 'Email already in use' }});
  const passwordHash = await hashPassword(password);
  const user = await prisma.user.create({ data: { email, passwordHash, displayName } });
  const { token, expiresAt } = await createSession(user.id, req.get('user-agent') ?? '', req.ip);
  res.cookie('od.sid', token, { httpOnly: true, sameSite: 'lax', secure: process.env.NODE_ENV === 'production', expires: expiresAt });
  res.status(201).json({ user: strip(user, ['passwordHash']) });
});
